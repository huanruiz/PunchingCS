简单的方法就是痛hashmap来存储然后dfs.
```
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None
from collections import defaultdict
class Solution:
    def findMode(self, root: TreeNode) -> List[int]:
        self.count = 0
        self.d = defaultdict(int)
        def dfs(root):
            if root == None:
                return
            self.d[root.val] += 1
            if self.d[root.val] > self.count:
                self.count = self.d[root.val]
            dfs(root.left)
            dfs(root.right)
        dfs(root)
        res = []
        for key in self.d:
            if self.d[key] == self.count:
                res.append(key)
        return res
```
