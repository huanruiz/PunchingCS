class Solution:
    def subsets(self, nums: List[int]) -> List[List[int]]:        
        res = [[]]
        for i in nums:
            res = res + [[i] + num for num in res]
        return res

回溯, 记住要用浅拷贝.
```
import copy
class Solution:
    def subsets(self, nums: List[int]) -> List[List[int]]:
        def dfs(start, path, nums):
            res.append(copy.copy(path))
            for i in range(start, len(nums)):
                path.append(nums[i])
                dfs(i + 1, path, nums)
                path.pop()
            return res

        res = []
        dfs(0, [], nums)
        return res
```
