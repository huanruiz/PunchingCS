通过dfs加回溯计算每条路径的值, 这里不能用大于剑指, 因为有负数.
```
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

import copy
class Solution:
    def pathSum(self, root: TreeNode, sum: int) -> List[List[int]]:
        self.sum = sum
        self.res = []
        def dfs(root, path, s):
            if root == None:
                return
            s += root.val
            if root.left == None and root.right == None:
                if s == self.sum:
                    self.res.append(copy.copy(path + [root.val]))
                return
            
            path.append(root.val)
            dfs(root.left, path, s)
            dfs(root.right, path, s)
            path.pop()
        dfs(root, [], 0)
        return self.res
```
