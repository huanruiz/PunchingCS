通过flag来判断是否为左节点, 再全部相加.
```
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def sumOfLeftLeaves(self, root: TreeNode) -> int:
        return self.dfs(root, False)
    
    def dfs(self, node, flag):
        if node == None:
            return 0  
        if flag and node.left == None and node.right == None:
            return node.val
        left = self.dfs(node.left, True)
        right = self.dfs(node.right, False)
        return left + right
```
