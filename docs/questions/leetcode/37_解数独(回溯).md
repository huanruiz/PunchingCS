回溯, 我用了三个表来记录每一行每一列哪些数字已经被使用了, 如果被其中一个list使用了, 则跳过. 或者如果这个点已经被数字占用了, 则看下一个. 暴力搜索, 从左到右, 从上到下.
```
class Solution:
    def solveSudoku(self, board: List[List[str]]) -> None:
        """
        Do not return anything, modify board in-place instead.
        """
        row = [[False for _ in range(9)] for _ in range(9)]
        colume = [[False for _ in range(9)] for _ in range(9)]
        box = [[[False for _ in range(3)] for _ in range(3)] for _ in range(9)]

        for i in range(9):
            for j in range(9):
                if not board[i][j] == '.':
                    num = int(board[i][j]) - 1
                    row[i][num] = True
                    colume[j][num] = True
                    box[num][i // 3][j // 3] = True
        
        self.dfs(board, row, colume, box, 0, 0)
    
    def dfs(self, board, row, colume, box, x, y):
        if y >= 9:
            y = 0
            x += 1
            if x >= 9:
                return True
        if board[x][y] == '.':
            for i in range(9):
                if not row[x][i] and not colume[y][i] and not box[i][x//3][y//3]:
                    board[x][y] = str(i + 1)
                    row[x][i], colume[y][i], box[i][x//3][y//3] = True, True, True
                    if self.dfs(board, row, colume, box, x, y + 1):
                        return True
                    board[x][y] = '.'
                    row[x][i] = False
                    colume[y][i] = False
                    box[i][x//3][y//3] = False
        else:
            return self.dfs(board, row, colume, box, x, y + 1)

        return False
```
